## 1. Reading Input

```rust
use std::io::{self, BufRead, Write, BufWriter, BufReader};

// ===== BASIC INPUT =====

// Single line
let mut line = String::new();
io::stdin().read_line(&mut line).unwrap();
let line = line.trim();                     // remove trailing \n

// Single integer
let n: i32 = line.trim().parse().unwrap();

// Two integers on same line
let mut line = String::new();
io::stdin().read_line(&mut line).unwrap();
let parts: Vec<i32> = line.trim().split_whitespace()
    .map(|x| x.parse().unwrap()).collect();
let (a, b) = (parts[0], parts[1]);

// Array of integers from single line
let arr: Vec<i32> = line.trim().split_whitespace()
    .map(|x| x.parse().unwrap()).collect();

// N lines of input
let n: usize = {
    let mut s = String::new();
    io::stdin().read_line(&mut s).unwrap();
    s.trim().parse().unwrap()
};
let mut lines = Vec::new();
for _ in 0..n {
    let mut s = String::new();
    io::stdin().read_line(&mut s).unwrap();
    lines.push(s.trim().to_string());
}

// 2D grid of integers
let mut grid: Vec<Vec<i32>> = Vec::new();
for _ in 0..n {
    let mut s = String::new();
    io::stdin().read_line(&mut s).unwrap();
    let row: Vec<i32> = s.trim().split_whitespace()
        .map(|x| x.parse().unwrap()).collect();
    grid.push(row);
}

// Character grid
let mut grid: Vec<Vec<char>> = Vec::new();
for _ in 0..n {
    let mut s = String::new();
    io::stdin().read_line(&mut s).unwrap();
    grid.push(s.trim().chars().collect());
}

// Comma separated
let vals: Vec<i32> = line.trim().split(',')
    .map(|x| x.trim().parse().unwrap()).collect();

// ===== FAST I/O (Competitive Programming) =====

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    let reader = BufReader::new(stdin.lock());

    for line in reader.lines() {
        let line = line.unwrap();
        // process
        writeln!(out, "{}", result).unwrap();
    }
}

// ===== INPUT HELPER (reusable across problems) =====

fn read_line() -> String {
    let mut s = String::new();
    io::stdin().read_line(&mut s).unwrap();
    s.trim().to_string()
}

fn read_int() -> i64 {
    read_line().parse().unwrap()
}

fn read_ints() -> Vec<i64> {
    read_line().split_whitespace()
        .map(|x| x.parse().unwrap()).collect()
}

// Read all stdin at once (fastest for competitive programming)
fn read_all() -> String {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    input
}
```

## 2. Output / Printing

```rust
// Basic
println!("{}", value);                      // with newline
print!("{}", value);                        // without newline
eprintln!("debug: {}", val);               // stderr (for debug)

// Formatted
println!("x={}, y={}", x, y);
println!("x={x}, y={y}");                  // inline var names (Rust 1.58+)
println!("{:>10}", val);                    // right-aligned, width 10
println!("{:0>5}", val);                    // zero-padded: "00042"
println!("{:.3}", 3.14159);                // 3 decimal places: "3.142"
println!("{:#b}", 42);                     // binary: "0b101010"
println!("{:#x}", 255);                    // hex: "0xff"
println!("{:?}", vec);                     // debug print

// Array as space-separated
let arr = vec![1, 2, 3];
let s: Vec<String> = arr.iter().map(|x| x.to_string()).collect();
println!("{}", s.join(" "));               // "1 2 3"
println!("{}", s.join(", "));              // "1, 2, 3"

// Print 2D grid
for row in &grid {
    let s: Vec<String> = row.iter().map(|x| x.to_string()).collect();
    println!("{}", s.join(" "));
}

// Debug print any collection
println!("{:?}", vec);                     // [1, 2, 3]
println!("{:#?}", map);                    // pretty-print with indentation

// Fast output with BufWriter
use std::io::{self, Write, BufWriter};
let stdout = io::stdout();
let mut out = BufWriter::new(stdout.lock());
writeln!(out, "{}", result).unwrap();
```

## 4. Math & Number Utilities

```rust
// Basic
x.max(y)                        // returns larger
x.min(y)                        // returns smaller
x.abs()                         // absolute value (signed types only)
x.pow(3)                        // integer power (u32 exponent)
(x as f64).sqrt() as i64        // integer square root
x.clamp(lo, hi)                 // clamp to range

// Integer division (truncates toward zero — same as C#)
7 / 2      // = 3
-7 / 2     // = -3  (NOT -4)

// Modulo (follows sign of dividend — same as C#)
7 % 3      // = 1
-7 % 3     // = -1   ← watch out
// Safe positive mod:
((x % m) + m) % m

// Division rounding up (ceiling division)
(a + b - 1) / b                  // ceiling(a / b) for positive a, b

// GCD
fn gcd(a: i64, b: i64) -> i64 {
    if b == 0 { a.abs() } else { gcd(b, a % b) }
}
fn lcm(a: i64, b: i64) -> i64 {
    a / gcd(a, b) * b            // divide first to avoid overflow
}

// Overflow-safe operations
x.checked_add(y)                // Option<T> — None on overflow
x.saturating_add(y)             // clamps at MAX/MIN
x.wrapping_add(y)               // wraps around
x.overflowing_add(y)            // (result, overflowed: bool)

// Power with modulo
fn mod_pow(mut base: i64, mut exp: i64, modulus: i64) -> i64 {
    let mut result = 1i64;
    base %= modulus;
    while exp > 0 {
        if exp & 1 == 1 {
            result = result * base % modulus;
        }
        exp >>= 1;
        base = base * base % modulus;
    }
    result
}

// Floating point
(x as f64).sqrt()
(x as f64).log2()
(x as f64).ceil() as i64
(x as f64).floor() as i64
f64::EPSILON                     // smallest representable difference

// Random (external crate — not available on LeetCode)
// For competitive programming: use simple LCG or xorshift
// LeetCode problems rarely need random
```

## 5. Arrays & Slices

```rust
// Fixed-size array (stack-allocated, size known at compile time)
let arr: [i32; 5] = [1, 2, 3, 4, 5];
let arr = [0i32; 100];                     // 100 zeros
let arr = [-1; 100];                       // 100 copies of -1

arr.len()
arr[0]                                      // first
arr[arr.len() - 1]                          // last

// Slice (borrowed view into array or Vec)
let s: &[i32] = &arr[1..4];               // [2, 3, 4] — borrowed, no allocation
let s: &[i32] = &arr[..3];                // first 3
let s: &[i32] = &arr[2..];                // from index 2

// Slice operations (work on &[T] and &mut [T])
s.len()
s.is_empty()
s.first()                                  // Option<&T>
s.last()                                   // Option<&T>
s.contains(&5)                             // O(n) linear scan
s.iter()                                   // iterate by reference
s.windows(3)                               // sliding window of size 3
s.chunks(2)                                // non-overlapping chunks
s.split(|&x| x == 0)                      // split by predicate

// Mutable slice operations
s.sort();                                  // in-place O(n log n)
s.sort_by(|a, b| b.cmp(a));              // descending
s.reverse();
s.swap(i, j);                             // swap two elements
s.fill(0);                                // fill all with 0
s.rotate_left(k);                         // rotate left by k positions
s.rotate_right(k);
```

## 6. Vec\<T\>

```rust
// Dynamic array — THE workhorse of Rust DSA
let mut v: Vec<i32> = Vec::new();
let mut v: Vec<i32> = Vec::with_capacity(1000);  // pre-allocate
let v = vec![0; n];                               // n zeros
let v = vec![-1; n];                              // n copies of -1
let v = vec![1, 2, 3, 4, 5];
let v = vec![vec![0; cols]; rows];                // 2D vec

// Add
v.push(item);                               // O(1) amortized — append to end
v.insert(index, item);                       // O(n) — shifts elements
v.extend_from_slice(&other);                 // O(k) append slice
v.extend(other.iter().cloned());             // append from iterator
v.append(&mut other);                        // move all from other (empties other)

// Remove
v.pop();                                     // O(1) — Option<T>, removes last
v.remove(index);                             // O(n) — shifts elements, returns removed
v.swap_remove(index);                        // O(1) — swaps with last, then pops (ORDER LOST)
v.retain(|&x| x > 0);                       // O(n) — remove elements not matching predicate
v.truncate(new_len);                         // O(k) — drop elements beyond new_len
v.clear();                                   // O(n) — drop all

// Access
v[0]                                         // panics on OOB
v.get(0)                                     // Option<&T> — safe
v.first()                                    // Option<&T>
v.last()                                     // Option<&T>
v.len()                                      // O(1)
v.is_empty()                                 // O(1)
v.contains(&item)                            // O(n) linear scan

// Slice operations (Vec auto-derefs to &[T])
v.sort();                                   // O(n log n)
v.sort_by(|a, b| b.cmp(a));               // descending
v.sort_by_key(|&x| x.abs());              // sort by key function
v.sort_unstable();                          // faster, not stable
v.dedup();                                  // remove CONSECUTIVE duplicates (sort first!)
v.reverse();
v.binary_search(&target);                   // Ok(idx) or Err(insertion_point)
v.partition_point(|&x| x < target);        // lower_bound equivalent
v.windows(k);                               // sliding windows of size k
v.chunks(k);                                // non-overlapping chunks
v.split(|&x| x == 0);                     // split by predicate

// Copy / Clone
let copy = v.clone();                       // deep clone
let slice = v[1..4].to_vec();             // sub-vec (allocates)

// Convert
let arr: [i32; 3] = v.try_into().unwrap(); // Vec → fixed array (if size matches)
let v: Vec<i32> = arr.to_vec();            // array → Vec

// Iteration
for x in &v { }                             // borrow
for x in &mut v { }                         // mutable borrow
for x in v { }                              // consume (moves ownership)
for (i, x) in v.iter().enumerate() { }     // index + value

// Stack-like usage (idiomatic in Rust DSA)
v.push(item);                               // push — O(1)
v.pop();                                     // pop — O(1), returns Option<T>
v.last();                                    // peek — O(1), returns Option<&T>
```

| Operation | Time |
|---|---|
| `push` | O(1) amortized |
| `pop` | O(1) |
| `v[i]` | O(1) |
| `insert(i, _)` | O(n) |
| `remove(i)` | O(n) |
| `swap_remove(i)` | O(1) |
| `contains` | O(n) |
| `sort` | O(n log n) |
| `binary_search` | O(log n) |
| `len` | O(1) |

## 7. Strings

```rust
// String = owned, heap-allocated, growable UTF-8
// &str   = borrowed string slice (view into String or literal)

let s = String::from("hello");
let s: &str = "hello";                     // string literal (static)
let s = "hello".to_string();               // &str → String

// Length
s.len()                                     // byte count, NOT char count
s.chars().count()                           // char count — O(n)
s.is_empty()

// Iteration (ALWAYS iterate chars, not bytes, for DSA)
for c in s.chars() { }
for (i, c) in s.chars().enumerate() { }
for b in s.bytes() { }                     // raw bytes

// Convert to char array (CRITICAL for index access)
let chars: Vec<char> = s.chars().collect();
chars[i]                                    // O(1) index access on chars
// s[i] DOES NOT WORK for char access — it's byte indexing

// Substring / Slicing (by byte index — dangerous with non-ASCII)
let sub = &s[0..5];                        // &str slice — panics if not char boundary
// SAFE alternative: use chars().skip(n).take(m)
let sub: String = s.chars().skip(1).take(3).collect(); // chars 1..4

// Search
s.contains("ell")                           // O(n)
s.starts_with("he")
s.ends_with("lo")
s.find("ll")                                // Option<usize> — byte index of first match
s.rfind("l")                                // from right

// Modification (creates NEW String)
s.to_uppercase()
s.to_lowercase()
s.trim()                                    // trim whitespace
s.trim_start()
s.trim_end()
s.replace("l", "r")                        // all occurrences
s.replacen("l", "r", 1)                    // first n occurrences

// Split & Join
let parts: Vec<&str> = s.split(' ').collect();
let parts: Vec<&str> = s.split_whitespace().collect();
let joined = parts.join(" ");              // Vec<&str> → String
let joined = parts.join(", ");

// Character checks
c.is_alphabetic()
c.is_ascii_alphabetic()
c.is_numeric()
c.is_ascii_digit()
c.is_alphanumeric()
c.is_ascii_lowercase()
c.is_ascii_uppercase()
c.is_whitespace()
c.to_ascii_lowercase()
c.to_ascii_uppercase()

// Char frequency (for lowercase English)
let mut freq = [0i32; 26];
for c in s.chars() {
    freq[(c as u8 - b'a') as usize] += 1;
}

// Reverse a string
let reversed: String = s.chars().rev().collect();

// Compare
s1 == s2                                    // value equality
s1.cmp(&s2)                                // lexicographic ordering
s1 < s2                                     // lexicographic comparison
```

| Operation | Time |
|---|---|
| `s.len()` (bytes) | O(1) |
| `s.chars().count()` | O(n) |
| `s.contains()` | O(n) |
| `s.find()` | O(n) |
| `chars[i]` (after collect) | O(1) |
| `s + &t` concatenation | O(n + m) |
| `s.split()` | O(n) |
| `s.replace()` | O(n) |

## 8. String Building

```rust
// Use String directly (it IS the builder)
let mut s = String::new();

s.insert(idx, 'c');                          // insert char at byte position

let s: String = vec!['a', 'b', 'c'].into_iter().collect();
let s: String = (0..10).map(|i| (b'a' + i) as char).collect();

let s = format!("{} + {} = {}", a, b, a + b);

let mut result = String::new();
for c in chars {
```

## 9. HashMap\<K,V\>

```rust
use std::collections::HashMap;

let mut map: HashMap<i32, i32> = HashMap::new();
let mut map: HashMap<i32, i32> = HashMap::with_capacity(1000);

// Insert / Update
map.insert(key, value);                     // returns Option<V> (old value)
map.entry(key).or_insert(default);         // insert if missing, return &mut V
*map.entry(key).or_insert(0) += 1;        // frequency counter — THE pattern

// Access
map.get(&key)                               // Option<&V>
map[&key]                                   // panics if missing
map.get(&key).copied()                     // Option<V> for Copy types
map.get(&key).unwrap_or(&default)          // with default
map.contains_key(&key)                     // O(1)

// Remove
map.remove(&key)                            // Option<V>
map.remove_entry(&key)                     // Option<(K, V)>
map.clear()

// Properties
map.len()
map.is_empty()
map.keys()                                  // iterator over &K
map.values()                                // iterator over &V
map.values_mut()                            // iterator over &mut V

// Iterate
for (k, v) in &map { }
for (k, v) in &mut map { }                // mutable values
for (k, v) in map { }                     // consume

// ========== DSA PATTERNS ==========

// Frequency counter
let mut freq: HashMap<i32, i32> = HashMap::new();
for &x in &arr {
    *freq.entry(x).or_insert(0) += 1;
}

// Group by key
let mut groups: HashMap<i32, Vec<i32>> = HashMap::new();
for &x in &arr {
    groups.entry(x % 3).or_insert_with(Vec::new).push(x);
}
// Or with or_default():
groups.entry(x % 3).or_default().push(x);

// Two-sum pattern
let mut seen: HashMap<i32, usize> = HashMap::new();
for (i, &x) in nums.iter().enumerate() {
    let complement = target - x;
    if let Some(&j) = seen.get(&complement) {
        return vec![j as i32, i as i32];
    }
    seen.insert(x, i);
}

// entry() API — full power
map.entry(key)
    .and_modify(|v| *v += 1)              // if exists: modify
    .or_insert(1);                          // if missing: insert

// Build from iterator
let map: HashMap<i32, i32> = vec![(1, 2), (3, 4)].into_iter().collect();
let map: HashMap<char, usize> = s.chars().enumerate().map(|(i, c)| (c, i)).collect();
```

| Operation | Time (avg) | Time (worst) |
|---|---|---|
| `insert` | O(1) | O(n) |
| `get` | O(1) | O(n) |
| `remove` | O(1) | O(n) |
| `contains_key` | O(1) | O(n) |
| `entry` | O(1) | O(n) |
| `len` | O(1) | O(1) |
| Iterate all | O(n) | O(n) |

```rust
use std::collections::HashSet;

let difference: HashSet<_> = a.difference(&b).cloned().collect();
let sym_diff: HashSet<_> = a.symmetric_difference(&b).cloned().collect();

// Set queries
a.is_subset(&b);
a.is_superset(&b);
a.is_disjoint(&b);                          // no common elements

// Iterate
for &item in &set { }

// ========== DSA PATTERNS ==========

// Duplicate detection
let mut seen = HashSet::new();
for &x in &arr {

// Visited tracking for grid BFS/DFS
let mut visited: HashSet<(usize, usize)> = HashSet::new();
visited.insert((row, col));
if visited.contains(&(nr, nc)) { continue; }

// O(1) lookup instead of O(n) contains on Vec
let lookup: HashSet<i32> = arr.iter().cloned().collect();
if lookup.contains(&target) { }
```

## 11. BTreeMap\<K,V\> / BTreeSet\<T\>

```rust
use std::collections::{BTreeMap, BTreeSet};

// Sorted map / set — B-tree backed, O(log n) operations
// Use when you need: sorted order, range queries, min/max by key

// BTreeMap
let mut map: BTreeMap<i32, i32> = BTreeMap::new();
map.insert(key, value);
map.get(&key);                              // Option<&V> — O(log n)
map.remove(&key);
map.contains_key(&key);

// Ordered access
map.first_key_value();                      // Option<(&K, &V)> — smallest key
map.last_key_value();                       // Option<(&K, &V)> — largest key
map.pop_first();                            // remove & return smallest
map.pop_last();                             // remove & return largest

// Range queries
map.range(lo..hi)                           // iterate keys in [lo, hi)
map.range(lo..=hi)                          // iterate keys in [lo, hi]
map.range(..hi)                             // iterate keys < hi
map.range(lo..)                             // iterate keys >= lo
for (k, v) in map.range(2..=5) { }

// Iterates in sorted key order (always)
for (k, v) in &map { }

// BTreeSet
let mut set: BTreeSet<i32> = BTreeSet::new();
set.insert(item);
set.remove(&item);
set.contains(&item);                        // O(log n)

set.first();                                // Option<&T> — smallest
set.last();                                 // Option<&T> — largest
set.pop_first();                            // remove smallest
set.pop_last();                             // remove largest
set.range(lo..hi)                           // range iterator
```

```rust
use std::rc::Rc;
use std::cell::RefCell;

// LeetCode TreeNode definition (memorize this)
#[derive(Debug, PartialEq, Eq)]
pub struct TreeNode {
    pub val: i32,
    pub left: Option<Rc<RefCell<TreeNode>>>,
    pub right: Option<Rc<RefCell<TreeNode>>>,
}

impl TreeNode {
    pub fn new(val: i32) -> Self {
        TreeNode { val, left: None, right: None }
    }
}

type TreeLink = Option<Rc<RefCell<TreeNode>>>;

// Traversal patterns
fn inorder(node: &TreeLink, result: &mut Vec<i32>) {
    if let Some(n) = node {
        let n = n.borrow();
        inorder(&n.left, result);
        result.push(n.val);
        inorder(&n.right, result);
    }
}

fn preorder(node: &TreeLink, result: &mut Vec<i32>) {
    if let Some(n) = node {
        let n = n.borrow();
        result.push(n.val);
        preorder(&n.left, result);
        preorder(&n.right, result);
    }
}

fn postorder(node: &TreeLink, result: &mut Vec<i32>) {
    if let Some(n) = node {
        let n = n.borrow();
        postorder(&n.left, result);
        postorder(&n.right, result);
        result.push(n.val);
    }
}

// Level order (BFS)
fn level_order(root: TreeLink) -> Vec<Vec<i32>> {
    let mut result = Vec::new();
    let mut queue = VecDeque::new();
    if let Some(r) = root { queue.push_back(r); }
    while !queue.is_empty() {
        let size = queue.len();
        let mut level = Vec::new();
        for _ in 0..size {
            let node = queue.pop_front().unwrap();
            let node = node.borrow();
            level.push(node.val);
            if let Some(ref left) = node.left {
                queue.push_back(Rc::clone(left));
            }
            if let Some(ref right) = node.right {
                queue.push_back(Rc::clone(right));
            }
        }
        result.push(level);
    }
    result
}

// Height
fn height(node: &TreeLink) -> i32 {
    match node {
        None => 0,
        Some(n) => {
            let n = n.borrow();
            1 + height(&n.left).max(height(&n.right))
        }
    }
}

// Access patterns
// node.borrow()            → immutable access
// node.borrow_mut()        → mutable access
// node.borrow().val        → get value
// node.borrow().left.clone() → get child (clone the Rc)
// Rc::clone(&node)         → cheap reference count clone
```

```rust
        let mut node = Box::new(ListNode::new(val));
        node.next = head;
        head = Some(node);
    }
    head
}
```

## 25. Custom Sorting & Ordering

```rust
use std::cmp::Ordering;

// Ord trait — the type knows how to compare itself
#[derive(Eq, PartialEq)]
struct Event { time: i32, kind: i32 }

impl Ord for Event {
    fn cmp(&self, other: &Self) -> Ordering {
        self.time.cmp(&other.time)
            .then(self.kind.cmp(&other.kind))
    }
}
impl PartialOrd for Event {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

// Reverse ordering (for min-heap from max-heap)
use std::cmp::Reverse;
heap.push(Reverse(val));

// Sort by closure
v.sort_by(|a, b| {
    a.start.cmp(&b.start)
        .then(a.end.cmp(&b.end))
});

// Sort by key
v.sort_by_key(|item| item.priority);
v.sort_by_key(|item| Reverse(item.priority));  // descending

// Multi-key sort as tuple (elegant)
v.sort_by_key(|item| (item.start, item.end));  // tuples compare lexicographically
v.sort_by_key(|item| (item.start, Reverse(item.end))); // start asc, end desc

// Ordering helpers
Ordering::Less      // a < b
Ordering::Equal     // a == b
Ordering::Greater   // a > b
ord.then(other_ord)                         // if Equal, use other
ord.then_with(|| a.y.cmp(&b.y))           // lazy evaluation
ord.reverse()                               // flip the ordering
```

```rust
// Union-Find
struct UnionFind {
    parent: Vec<usize>,
    rank: Vec<usize>,
}
impl UnionFind {
    fn new(n: usize) -> Self {
        Self { parent: (0..n).collect(), rank: vec![0; n] }
    }
    fn find(&mut self, x: usize) -> usize {
        if self.parent[x] != x {
            self.parent[x] = self.find(self.parent[x]);
        }
        self.parent[x]
    }
    fn union(&mut self, x: usize, y: usize) -> bool {
        let (px, py) = (self.find(x), self.find(y));
        if px == py { return false; }
        if self.rank[px] < self.rank[py] {
            self.parent[px] = py;
        } else if self.rank[px] > self.rank[py] {
            self.parent[py] = px;
        } else {
            self.parent[py] = px;
            self.rank[px] += 1;
        }
        true
    }
}
```

```rust
// Most DSA in Rust = pass &[T] (borrowed slices), return owned values
```

## 28. Gotchas & Traps

| Trap | Problem | Fix |
|---|---|---|
| `usize` underflow `i - 1` when `i == 0` | Panics in debug, wraps in release | Check `i > 0` first, or use `i32` for loop var |
| `s[i]` on String | Byte indexing, panics on non-char boundary | Collect to `Vec<char>`, then index |
| `HashMap` iteration order random | Can't rely on insertion order | Use `BTreeMap` for sorted, or collect + sort |
| `v.sort()` with floats | `f64` doesn't implement `Ord` | Use `sort_by(\|a,b\| a.partial_cmp(b).unwrap())` |
| Integer overflow `(lo + hi) / 2` | Can overflow for large values | Use `lo + (hi - lo) / 2` |
| `.clone()` in hot loop | Expensive for large data | Use references or indices |
| `collect::<Vec<_>>()` type inference fails | Compiler can't infer element type | Annotate: `collect::<Vec<i32>>()` |
| Negative mod: `-7 % 3 = -1` | Not `2` like Python | `((x % m) + m) % m` |
| `BinaryHeap` is max-heap | Pops largest, not smallest | Wrap with `Reverse(x)` |
| Moving out of `Vec` in loop | Can't move and iterate | Use `.iter()` for borrows, `.remove()` / `.swap_remove()` to extract |
| Mutable borrow + immutable borrow | Can't have both at once | Clone, use indices, or restructure |
| `Option::unwrap()` in contests | Panics if None | Fine in contests, but use `?` or `if let` in production |
| `vec![vec![]; n]` with non-Clone | Inner type must implement Clone | Use `(0..n).map(\|_\| Vec::new()).collect()` |
| Integer division truncates | `-7 / 2 = -3` (not `-4`) | Same as C#, different from Python |
| `HashSet<Vec<i32>>` | Vec doesn't implement Hash | Convert to tuple, sort + convert to String, or use BTreeSet |
| `entry().or_insert_with()` vs `or_insert()` | `or_insert(Vec::new())` creates Vec even if key exists | Use `or_default()` or `or_insert_with(Vec::new)` for lazy init |
