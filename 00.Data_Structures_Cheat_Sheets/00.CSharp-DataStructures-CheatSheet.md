## 4. Math & Number Utilities

```csharp
Math.Max(a, b)
Math.Min(a, b)
Math.Abs(x)
Math.Pow(base, exp)           // returns double
Math.Sqrt(x)                  // returns double
Math.Ceiling(x)               // rounds up (double → double)
Math.Floor(x)                 // rounds down
Math.Log(x)                   // natural log
Math.Log2(x)                  // base-2 log (C# 10+)
Math.Clamp(val, min, max)

// Integer division (truncates toward zero)
7 / 2    // = 3
-7 / 2   // = -3  (NOT -4, unlike Python)

// Modulo (follows sign of dividend)
7 % 3    // = 1
-7 % 3   // = -1   ← watch out
// Safe positive mod:
int mod = ((x % m) + m) % m;

// GCD (C# doesn't have built-in before .NET 7)
static int Gcd(int a, int b) => b == 0 ? a : Gcd(b, a % b);
static int Lcm(int a, int b) => a / Gcd(a, b) * b; // divide first to avoid overflow

// Overflow detection
checked { int result = a + b; }  // throws OverflowException
unchecked { int result = a + b; } // wraps silently (default)

// Power with modulo (for large numbers)
static long ModPow(long baseVal, long exp, long mod)
{
    long result = 1;
    baseVal %= mod;
    while (exp > 0)
    {
        if ((exp & 1) == 1) result = result * baseVal % mod;
        exp >>= 1;
        baseVal = baseVal * baseVal % mod;
    }
    return result;
}

// Random
var rng = new Random();
int r = rng.Next(0, 10);      // [0, 10)
int r = rng.Next(10);         // [0, 10)
```

## 9. Dictionary\<K,V\>

```csharp
// Hash map — THE most-used DS in DSA
var map = new Dictionary<int, int>();
var map = new Dictionary<int, int>(capacity);
var map = new Dictionary<string, List<int>>();

// Add / Update
map[key] = value;                           // insert or overwrite
map.Add(key, value);                        // throws if key exists
map.TryAdd(key, value);                     // returns false if exists

// Access
int val = map[key];                         // throws if missing
bool found = map.TryGetValue(key, out int val); // safe — returns false if missing
bool exists = map.ContainsKey(key);         // O(1)
bool hasVal = map.ContainsValue(val);       // O(n) — linear scan!

// Remove
map.Remove(key);                            // O(1)
map.Remove(key, out int removed);          // remove & get value
map.Clear();

// Properties
map.Count
map.Keys                                    // ICollection<K>
map.Values                                  // ICollection<V>

// Iterate
foreach (var kvp in map)                    // KeyValuePair<K,V>
    Console.WriteLine($"{kvp.Key}: {kvp.Value}");
foreach (var (key, val) in map)             // deconstruct (C# 7+)
    Console.WriteLine($"{key}: {val}");
foreach (int key in map.Keys) { }
foreach (int val in map.Values) { }

// ========== DSA PATTERNS ==========

// Frequency counter (THE most common pattern)
var freq = new Dictionary<int, int>();
foreach (int x in arr)
{
    freq.TryGetValue(x, out int count);
    freq[x] = count + 1;
}
// Or with CollectionsMarshal (C# 10+, fastest):
// Or simply:
if (freq.ContainsKey(x)) freq[x]++;
else freq[x] = 1;

// GetValueOrDefault (cleaner)
int count = freq.GetValueOrDefault(key, 0);

// Group items by key
var groups = new Dictionary<int, List<int>>();
foreach (int x in arr)
{
    if (!groups.ContainsKey(x % 3))
        groups[x % 3] = new List<int>();
    groups[x % 3].Add(x);
}
// Or with TryGetValue:
if (!groups.TryGetValue(key, out var list))
{
    list = new List<int>();
    groups[key] = list;
}
list.Add(value);

// Two-sum pattern
var seen = new Dictionary<int, int>(); // value → index
for (int i = 0; i < nums.Length; i++)
{
    int complement = target - nums[i];
    if (seen.TryGetValue(complement, out int j))
        return new[] { j, i };
    seen[nums[i]] = i;
}
```

## 11. SortedSet\<T\> / SortedDictionary\<K,V\>

```csharp
// Red-black tree — sorted order, O(log n) operations
// Use when you need: min/max, floor/ceiling, ordered iteration

// SortedSet<T>
var ss = new SortedSet<int>();
ss.Add(5);
ss.Remove(5);
ss.Contains(5);                             // O(log n)
ss.Min;                                     // O(log n) — smallest
ss.Max;                                     // O(log n) — largest
ss.Count;

// Range operations
var view = ss.GetViewBetween(lo, hi);       // O(log n + k) — SortedSet of items in [lo, hi]
ss.Reverse();                               // iterate in descending order

// SortedDictionary<K,V>
var sd = new SortedDictionary<int, string>();
sd[key] = value;
sd.Remove(key);
sd.ContainsKey(key);                        // O(log n)
// Iterates in sorted key order
foreach (var (k, v) in sd) { }

// SortedList<K,V> — array-backed sorted map

var sl = new SortedList<int, int>();

int keyAt = sl.Keys[0];                    // O(1) — indexed access
int valAt = sl.Values[0];                  // O(1)
```

## 12. Stack\<T\>

```csharp
var stack = new Stack<int>();
var stack = new Stack<int>(capacity);
var stack = new Stack<int>(collection);     // pushes in order (last = top)

stack.Push(item);                           // O(1)
int top = stack.Pop();                      // O(1) — throws if empty

stack.Count;
stack.Contains(item);                       // O(n) — linear scan
stack.Clear();

// Convert
int[] arr = stack.ToArray();                // top of stack = arr[0]

// ========== DSA PATTERNS ==========

// Monotonic stack (next greater element)
var stk = new Stack<int>();                // stores indices
int[] result = new int[n];
Array.Fill(result, -1);
for (int i = 0; i < n; i++)
{
    while (stk.Count > 0 && nums[stk.Peek()] < nums[i])
        result[stk.Pop()] = nums[i];
    stk.Push(i);
}

// Valid parentheses
var stk = new Stack<char>();
foreach (char c in s)
{
    if (c == '(' || c == '[' || c == '{') stk.Push(c);
    else
    {
        if (stk.Count == 0) return false;
        char open = stk.Pop();
        if ((c == ')' && open != '(') ||
            (c == ']' && open != '[') ||
            (c == '}' && open != '{')) return false;
    }
}
return stk.Count == 0;

// DFS iterative (replace recursion)
var stk = new Stack<int>();
stk.Push(start);
var visited = new HashSet<int>();
while (stk.Count > 0)
{
    int node = stk.Pop();
    if (!visited.Add(node)) continue;
    foreach (int next in graph[node])
        stk.Push(next);
}
```

```csharp
visited[start] = true;

while (queue.Count > 0)
{
    int size = queue.Count;                 // process level by level
    for (int i = 0; i < size; i++)
    {
        int node = queue.Dequeue();
        foreach (int next in graph[node])
        {
            if (!visited[next])
            {
                visited[next] = true;
                queue.Enqueue(next);
            }
        }
    }
    level++;
}

// BFS on grid
var queue = new Queue<(int row, int col)>();
int[][] dirs = { new[]{0,1}, new[]{0,-1}, new[]{1,0}, new[]{-1,0} };
queue.Enqueue((startR, startC));
visited[startR, startC] = true;
while (queue.Count > 0)
{
    var (r, c) = queue.Dequeue();
    foreach (var d in dirs)
    {
        int nr = r + d[0], nc = c + d[1];
        if (nr >= 0 && nr < rows && nc >= 0 && nc < cols
            && !visited[nr, nc] && grid[nr][nc] != '#')
        {
            visited[nr, nc] = true;
            queue.Enqueue((nr, nc));
        }
    }
}
```

## 15. PriorityQueue\<E,P\>

```csharp
// .NET 6+ built-in — MIN-HEAP by default (dequeues lowest priority first)
var pq = new PriorityQueue<int, int>();     // <Element, Priority>

// Add
pq.Enqueue(element, priority);              // O(log n)

// Remove
int elem = pq.Dequeue();                   // O(log n) — removes lowest priority
int elem = pq.Peek();                      // O(1) — view without removing
bool ok = pq.TryDequeue(out int elem, out int priority);
bool ok = pq.TryPeek(out int elem, out int priority);

pq.Count;
pq.EnqueueDequeue(element, priority);      // O(log n) — enqueue then dequeue min (optimized)

// ========== MAX-HEAP: negate priorities ==========
pq.Enqueue(element, -priority);            // negate to simulate max-heap

// ========== DSA PATTERNS ==========

// Dijkstra
var pq = new PriorityQueue<int, int>();    // (node, distance)
int[] dist = new int[n];
Array.Fill(dist, int.MaxValue);
dist[start] = 0;
pq.Enqueue(start, 0);
while (pq.Count > 0)
{
    pq.TryDequeue(out int node, out int d);
    if (d > dist[node]) continue;          // stale entry
    foreach (var (next, weight) in graph[node])
    {
        int newDist = d + weight;
        if (newDist < dist[next])
        {
            dist[next] = newDist;
            pq.Enqueue(next, newDist);
        }
    }
}

// Top K largest (use min-heap of size k)
var pq = new PriorityQueue<int, int>();
foreach (int x in nums)
{
    pq.Enqueue(x, x);                     // min-heap by value
    if (pq.Count > k) pq.Dequeue();       // evict smallest
}
// pq now contains k largest elements

// Top K smallest (use max-heap of size k — negate priority)
pq.Enqueue(x, -x);

// Merge K sorted lists
var pq = new PriorityQueue<(int listIdx, int elemIdx), int>();
for (int i = 0; i < lists.Length; i++)
    if (lists[i].Length > 0)
        pq.Enqueue((i, 0), lists[i][0]);
while (pq.Count > 0)
{
    var (li, ei) = pq.Dequeue();
    result.Add(lists[li][ei]);
    if (ei + 1 < lists[li].Length)
        pq.Enqueue((li, ei + 1), lists[li][ei + 1]);
}
```

| Operation | Time |
|---|---|
| `Enqueue` | O(log n) |
| `Dequeue` | O(log n) |
| `Peek` | O(1) |
| `Count` | O(1) |

|---|---|---|
| `a - b` comparator overflow | `int.MaxValue - (-1)` overflows | Use `a.CompareTo(b)` |
| String `+` in loop | O(n²) — creates new string each time | Use `StringBuilder` |
| `List.Contains()` in loop | O(n) per call → O(n²) total | Use `HashSet<T>` |
| `Dictionary[key]` when missing | Throws `KeyNotFoundException` | Use `TryGetValue` |
| `int` overflow in `mid = (lo + hi) / 2` | When lo + hi > int.MaxValue | Use `lo + (hi - lo) / 2` |
| `-7 / 2 = -3` not `-4` | C# truncates toward zero | Use `Math.Floor((double)a / b)` for floor division |
| `-7 % 3 = -1` not `2` | Mod follows dividend sign | `((x % m) + m) % m` |
| `Array.Sort` is unstable | Equal elements may reorder | Use LINQ `.OrderBy()` for stable |
| `Stack.ToArray()` order | Top of stack = index 0 | Reverse if you need bottom-first |
| Value type in `Dictionary` | `map[key]++` works; but modifying struct fields needs reassign | Copy out, modify, copy back |
| LINQ lazy evaluation | `.Where().Count()` evaluates twice if called twice | Materialize with `.ToList()` |
| `PriorityQueue` is min-heap | Dequeues smallest priority first | Negate for max-heap |
| `HashSet<int[]>` | Arrays use reference equality | Use `HashSet<(int,int)>` tuples |
| `default(int)` is 0 | `TryGetValue` returns 0 for missing int keys | Fine for counting, but know it |
